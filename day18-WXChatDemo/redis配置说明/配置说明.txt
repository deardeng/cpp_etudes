
1 创建redis配置目录
 
 /home/zack/wxchat/redisconf

2 创建数据目录  

/home/zack/wxchat/redisdata

3 运行容器

docker run -p 6380:6380 --name wxredis --privileged=true -v /home/zack/wxchat/redisconf/redis.conf:/etc/redis/redis.conf -v /home/zack/wxchat/redisdata:/data -d redis redis-server /etc/redis/redis.conf --appendonly yes

4 进入容器
docker exec -it wxredis /bin/bash

5 密码登录
redis-cli -p 6380
